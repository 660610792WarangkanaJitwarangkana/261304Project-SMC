R1 = 5           // ตัวตั้ง
R2 = 3           // ตัวคูณ  
R4 = -1          // สำหรับลดค่า
R5 = 0           // ผลลัพธ์

while R2 != 0:
    R5 = R5 + R1  // บวกตัวตั้ง
    R2 = R2 + R4  // ลดตัวคูณลง 1
end while

halt

ตัวอย่างการทำงาน 5*3
รอบที่ 1: R2=3, R5=0+5=5
รอบที่ 2: R2=2, R5=5+5=10  
รอบที่ 3: R2=1, R5=10+5=15
รอบที่ 4: R2=0 → ออกจากลูป
ผลลัพธ์: R5 = 15
